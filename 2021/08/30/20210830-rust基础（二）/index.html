    绝对路径和相对路径

    mod front_of_house {
    mod hosting {
        fn add_to_waitlist() {}
    }
}

pub fn eat_at_restaurant() {
    // Absolute path，被定义在同一crate中
    crate::front_of_house::hosting::add_to_waitlist();

    // Relative pathfront_of_house与 eat_at_restaurant 定义在同一层级
    front_of_house::hosting::add_to_waitlist();
}

    pub关键字：

    super关键字：从父模块开始的相对路径；

    公有结构体的字段可以设置部分公有，部分私有；

    use：关键字将名称引入作用域；

    mod front_of_house {
    pub mod hosting {
        pub fn add_to_waitlist() {}
    }
}

use crate::front_of_house::hosting;
pub fn eat_at_restaurant() {
    hosting::add_to_waitlist();
    hosting::add_to_waitlist();
    hosting::add_to_waitlist();
}

    使用 & 和 [] 返回一个引用；或者使用 get 方法以索引作为参数来返回一个 Option<&T>

    // 访问vec中的元素
let v = vec![1,2,3,4,5];

let third:&i32 = &v[2];

match v.get(2){
 Some(third)=>println!("hhh"),
 None => println!("h"),
}

    // 遍历vector中的元素
let v = vec![100, 32, 57];
for i in &v {
    println!("{}", i);

for i in &mut v {
    *i += 50;

    字符串存储utf-8编码文本

    &str:引用

    String:可增长的、可变的、有所有权的

    let data = "initial contents";

let s =data.to_string();

let s ="initial contents".to_string();

let s =String::from("initial contents");

// push获取引用，不取得所有权
let mut s = String::from("foo");
s.push_str("bar");

// 单独一个字符
s.push("l");

// + 运算符与format！
let s1 = String::from("Hello, ");
let s2 = String::from("world!");
let s3 = s1 + &s2;
// 注意 s1 被移动了，不能继续使用

// 注意s1的控制权没有了，只剩下s
let s1 = String::from("tic");
let s2 = String::from("tac");
let s3 = String::from("toe");
let s = s1 + "-" + &s2 + "-" + &s3;
// 不会获得s1的所有权
let s = format!("{}-{}-{}", s1, s2, s3);
